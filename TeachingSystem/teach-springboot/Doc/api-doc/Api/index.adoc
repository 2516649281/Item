[source,HTTP ]
----
POST /file/load HTTP/1.1
Content-Type: application/json

[ 0 ]
----

----

  "statue" : "200",
  "message" : "请求已完成!",
  "data" : {
    "0" : 0
  },
  "pageSize" : "0L"
